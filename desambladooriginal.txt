                0160 ; These macros arecalled by the actual I/O macros
                0170 ; to perform the rudimentary register load functions.
                0180 ;
                0190 ;
                0200 ; MACRO:  @CH
                0210 ;
                0220 ; Loads IOCB number (parameter 1) into X register.
                0230 ;
                0240 ; If parameter value is 0 to 7, immediate channel number
                0250 ;   is assumed.
                0260 ;
                0270 ; If parameter value is > 7 then a memory location
                0280 ;   is assumed to contain the channel number.
                0290 ;
                0300     .MACRO @CH
                0310     .IF %1>7
                0320       LDA %1
                0330       ASL A
                0340       ASL A
                0350       ASL A
                0360       ASL A
                0370       TAX
                0380       .ELSE
                0390       LDX #%1*16
                0400       .ENDIF
                0410     .ENDM
                0420 ;
                0430 ;
                0440 ; MACRO:  @CV
                0450 ;
                0460 ; Loads Constant or Value into accumultor (A-register)
                0470 ;
                0480 ; If value of parameter 1 is 0-255, @CV
                0490 ; assumes it's an (immediate) constant.
                0500 ;
                0510 ; Otherwise the value is assumed to
                0520 ; be a memory location (non-zero page).
                0530 ;
                0540 ;
                0550 ;
                0560     .MACRO @CV
                0570     .IF %1<256
                0580       LDA #%1
                0590       .ELSE
                0600       LDA %1
                0610       .ENDIF
                0620     .ENDM
                0630 ;
                0640 ;
                0650 ;
                0660 ;
                0670 ; MACRO:  @FL
                0680 ;
                0690 ; @FL is used to establish a filespec (file name)
                0700 ;
                0710 ; If a literal string is passed, @FL will
                0720 ; generate the string in line, jump
                0730 ; around it, and place its address
                0740 ; in the IOCB pointed to by the X-register.
                0750 ;
                0760 ; If a non-zero page label is passed
                0770 ; the MACRO assumes it to be the label
                0780 ; of a valid filespec and uses it instead.
                0790 ;
                0800 ;
                0810 ;
                0820     .MACRO @FL
                0830     .IF %1<256
                0840       JMP *+%1+4
                0850 @F    .BYTE %$1,0
                0860       LDA # <@F
                0870       STA ICBADR,X
                0880       LDA # >@F
                0890       STA ICBADR+1,X
                0900       .ELSE
                0910       LDA # <%1
                0920       STA ICBADR,X
                0930       LDA # >%1
                0940       STA ICBADR+1,X
                0950       .ENDIF
                0960     .ENDM
                0970     .PAGE "   XIO macro"
                0980 ;
                0990 ; MACRO:  XIO
                1000 ;
                1010 ;  FORM:  XIO cmd,ch[,aux1,aux2][,filespec]
                1020 ;
                1030 ; ch is given as in the @CH macro
                1040 ; cmd, aux1, aux2 are given as in the @CV macro
                1050 ; filespec is given as in the @FL macro
                1060 ;
                1070 ; performs familiar XIO operations with/for OS/A+
                1080 ;
                1090 ; If aux1 is given, aux2 must also be given
                1100 ; If aux1 and aux2are omitted, they are set to zero
                1110 ; If the filespec is omitted, "S:" is assumed
                1120 ;
                1130     .MACRO XIO
                1140     .IF %0<2 .OR %0>5
                1150       .ERROR "XIO:wrong number of arguments"
                1160       .ELSE
                1170        @CH  %2
                1180        @CV  %1
                1190       STA ICCOM,X ; COMMAND
                1200       .IF %0>=4
                1210          @CV  %3
                1220         STA ICAUX1,X
                1230          @CV  %4
                1240         STA ICAUX2,X
                1250         .ELSE
                1260         LDA #0
                1270         STA ICAUX1,X
                1280         STA ICAUX2,X
                1290         .ENDIF
                1300       .IF %0=2 .OR%0=4
                1310          @FL  "S:"
                1320         .ELSE
                1330 @@IO    .=  %0
                1340          @FL  %$(@@IO)
                1350         .ENDIF
                1360       JSR CIO
                1370       .ENDIF
                1380     .ENDM
                1390     .PAGE "   OPEN macro"
                1400 ;
                1410 ; MACRO:  OPEN
                1420 ;
                1430 ;  FORM:  OPEN ch,aux1,aux2,filespec
                1440 ;
                1450 ; ch is given as in the @CH macro
                1460 ; aux1 and aux2 are given as in the @CV macro
                1470 ; filespec is given as in the @FL macro
                1480 ;
                1490 ; will attempt to open the given file name on
                1500 ; the given channel, using the open "modes"
                1510 ; specified by aux1 and aux2
                1520 ;
                1530     .MACRO OPEN
                1540     .IF %0<>4
                1550       .ERROR "OPEN: wrong number of arguments"
                1560       .ELSE
                1570       .IF %4<256
                1580          XIO  COPN,%1,%2,%3,%$4
                1590         .ELSE
                1600          XIO  COPN,%1,%2,%3,%4
                1610         .ENDIF
                1620       .ENDIF
                1630     .ENDM
                1640     .PAGE "   BGET and BPUT macros"
                1650 ;
                1660 ; MACROS: BGET and BPUT
                1670 ;
                1680 ;   FORM: BGET ch,buf,len
                1690 ;         BPUT ch,buf,len
                1700 ;
                1710 ; ch is given as in the @CH macro
                1720 ; len is ALWAYS assumed to be an immediate
                1730 ;   and actual value...never a memory address
                1740 ; buf must be the address of an appropriate
                1750 ;   buffer in memory
                1760 ;
                1770 ; puts or gets length bytes to/from the
                1780 ;   specified buffer, uses binary read/write
                1790 ;
                1800 ;
                1810 ; first: a common macro
                1820 ;
                1830     .MACRO @GP
                1840      @CH  %1
                1850     LDA #%4
                1860     STA ICCOM,X
                1870     LDA # <%2
                1880     STA ICBADR,X
                1890     LDA # >%2
                1900     STA ICBADR+1,X
                1910     LDA # <%3
                1920     STA ICBLEN,X
                1930     LDA # >%3
                1940     STA ICBLEN+1,X
                1950     JSR CIO
                1960     .ENDM
                1970 ;
                1980     .MACRO BGET
                1990     .IF %0<>3
                2000       .ERROR "BGET: wrong number of parameters"
                2010       .ELSE
                2020       LDX #%1*16
                2030       LDA #$07
                2040       STA ICCOM,X
                2050       LDA # <%2
                2060       STA ICBADR,X
                2070       LDA # >%2
                2080       STA ICBADR+1,X
                2090       LDA %3
                2100       STA ICBLEN,X
                2110       LDA %3+1
                2120       STA ICBLEN+1,X
                2130       JSR CIO
                2140       .ENDIF
                2150     .ENDM
                2160 ;
                2170     .MACRO BPUT
                2180     .IF %0<>3
                2190       .ERROR "BPUT: wrong number of parameters"
                2200       .ELSE
                2210        @GP  %1,%2,%3,CPBINR
                2220       .ENDIF
                2230     .ENDM
                2240 ;
                2250     .PAGE "   PRINT macro"
                2260 ;
                2270 ; MACRO:  PRINT
                2280 ;
                2290 ;  FORM:  PRINT ch[,buffer[,length]]
                2300 ;
                2310 ; ch is as given in @CH macro
                2320 ; if no buffer, prints just a RETURN
                2330 ; if no length given, 255 assumed
                2340 ;
                2350 ; used to print text.  To print text without RETURN,
                2360 ; length must be given.  See OS/A+ manual
                2370 ;
                2380 ; EXCEPTION: second parameter may be a literal
                2390 ;  string (e.g., PRINT 0,"test"), in which
                2400 ;  case the length(if given) is ignored.
                2410 ;
                2420     .MACRO PRINT
                2430     .IF %0<1 .OR %0>3
                2440       .ERROR "PRINT: wrong number of parameters"
                2450       .ELSE
                2460       .IF %0>1
                2470         .IF %2<128
                2480           JMP *+4+%2
                2490 @IO       .BYTE %$2,$9B
                2500            @GP  %1,@IO,%2+1,CPTXTR
                2510           .ELSE
                2520           .IF %0=2
                2530              @GP  %1,%2,255,CPTXTR
                2540             .ELSE
                2550              @GP  %1,%2,%3,CPTXTR
                2560             .ENDIF
                2570           .ENDIF
                2580         .ELSE
                2590         JMP *+4
                2600 @IO     .BYTE $9B
                2610          @GP  %1,@IO,1,CPTXTR
                2620         .ENDIF
                2630       .ENDIF
                2640     .ENDM
                2650 ;
                2660     .PAGE "   INPUT macro"
                2670 ;
                2680 ; MACRO:  INPUT
                2690 ;
                2700 ;  FORM:  INPUT ch,buf,len
                2710 ;
                2720 ; ch is given as in the @CH macro
                2730 ; buf MUST be a proper buffer address
                2740 ; len may be omitted, in which case 255 is assumed
                2750 ;
                2760 ; gets a line of text input to the given
                2770 ;   buffer, maximum of length bytes
                2780 ;
                2790     .MACRO INPUT
                2800     .IF %0<2 .OR %0>3
                2810       .ERROR "INPUT: wrong number of parameters"
                2820       .ELSE
                2830       .IF %0=2
                2840          @GP  %1,%2,255,CGTXTR
                2850         .ELSE
                2860          @GP  %1,%2,%3,CGTXTR
                2870         .ENDIF
                2880       .ENDIF
                2890     .ENDM
                2900     .PAGE "   CLOSE macro"
                2910 ;
                2920 ; MACRO:  CLOSE
                2930 ;
                2940 ;  FORM:  CLOSE ch
                2950 ;
                2960 ; ch is given as in the @CH macro
                2970 ;
                2980 ; closes channel ch
                2990 ;
                3000     .MACRO CLOSE
                3010     .IF %0<>1
                3020       .ERROR "CLOSE: wrong number of parameters"
                3030       .ELSE
                3040        @CH  %1
                3050       LDA #CCLOSE
                3060       STA ICCOM,X
                3070       JSR CIO
                3080       .ENDIF
                3090     .ENDM
